    headfont=\bfseries\color{black!70!black}, bodyfont=\normalfont,
    mdframed={
        linewidth=1pt,
        rightline=true, topline=true, bottomline=true,
        linecolor=black, backgroundcolor=black!0!white,
    }
]{thmbox}


\declaretheoremstyle[
    headfont=\bfseries\color{black!3}, bodyfont=\normalfont,
    mdframed={
        linewidth=1pt,
        rightline=false, topline=false, bottomline=false,
        linecolor=black, backgroundcolor=black!0!white,align=center,
    }
]{thmtikz}


\declaretheoremstyle[
    headfont=\bfseries\color{black!70!black}, bodyfont=\normalfont,
    numbered=no,
    mdframed={
        linewidth=0pt,
        rightline=false, topline=false,leftline=false, bottomline=false,
        linecolor=black, backgroundcolor=black!0!white,
    },
    qed=\qedsymbol
]{thmproofbox}

\declaretheorem[sibling=theorem,style=thmbox, name=Lema]{lemma}
\declaretheorem[sibling=theorem,style=thmbox, name=Corolario]{corollary}

\newcommand{\hugequote}{%
	{\fontsize{75}{75}\selectfont%
	\hspace*{-.560em}\color{ctcolorgraylight}%
	\textit{\glqq}%
	\vskip -.26em}%
}

\newcommand{\cleanchapterquote}[3]{%
	\begin{minipage}{.865\textwidth}%
		\begin{flushright}
			\begin{minipage}{.65\textwidth}%
				\begin{flushleft}
					{\hugequote}\textit{#1}
				\end{flushleft}
	    		\begin{flushright}
    				--- \textbf{#2} \\
    				#3
		    	\end{flushright}
		    \end{minipage}%
		\end{flushright}
	\end{minipage}%
	\bigskip
}
